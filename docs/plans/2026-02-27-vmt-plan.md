# vmt Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a CLI tool that manages ephemeral QEMU/libvirt VMs for visual integration testing of Wayland applications, with screenshot diffing and SPICE manual access.

**Architecture:** Python CLI (`vmt`) uses libvirt to manage VMs booted from cloud images with cloud-init provisioning. SSH for programmatic access, SPICE for manual access. Screenshot pipeline: grim inside VM, SCP back, SSIM diff against references. Two-level TOML manifests — VM definitions owned by vmt, test scenarios owned by the project under test.

**Tech Stack:** Python 3.11+, libvirt-python, paramiko, scikit-image, QEMU/KVM, cloud-init

### Task 1: Host Setup

**Files:**
- None (system package installation)

**Step 1: Install QEMU, libvirt, and supporting tools**

```bash
sudo pacman -S --noconfirm qemu-full libvirt virt-viewer dnsmasq cloud-image-utils
```

`qemu-full` provides `qemu-system-x86_64` and `qemu-img`. `libvirt` provides `virsh` and the libvirt daemon. `virt-viewer` provides `remote-viewer` for SPICE. `dnsmasq` is needed by libvirt's default NAT network. `cloud-image-utils` provides `cloud-localds` for generating cloud-init ISOs.

**Step 2: Enable libvirtd and add user to group**

```bash
sudo systemctl enable --now libvirtd
sudo usermod -aG libvirt ifit
```

Note: the group change requires a new login session (or `newgrp libvirt`) to take effect. The subagent should verify with `groups` after.

**Step 3: Verify the default NAT network exists**

```bash
newgrp libvirt  # or the session may already have it
virsh net-list --all
```

If `default` network is not listed or not active:

```bash
virsh net-start default
virsh net-autostart default
```

**Step 4: Verify KVM is available**

```bash
ls -la /dev/kvm
```

Expected: character device exists and is accessible. If not, KVM modules may need loading (`modprobe kvm_intel` or `kvm_amd`).

```python
"""Parse VM and test manifest TOML files."""
from __future__ import annotations

import tomllib
from pathlib import Path

_VM_DEFAULTS = {
    "memory": 2048,
    "cpus": 2,
    "disk": 10,
}

_REQUIRED_VM_FIELDS = ["name", "image"]
_REQUIRED_SECTIONS = ["vm", "provision", "ssh"]


def load_vm_manifest(path: Path) -> dict:
    """Load and validate a VM manifest TOML file.

    Raises ValueError if required sections or fields are missing.
    """
    with open(path, "rb") as f:
        data = tomllib.load(f)

    for section in _REQUIRED_SECTIONS:
        if section not in data:
            raise ValueError(f"Missing required section [{section}] in {path}")

    for field in _REQUIRED_VM_FIELDS:
        if field not in data["vm"]:
            raise ValueError(
                f"Missing required field '{field}' in [vm] section of {path}"
            )

    # Apply defaults
    for key, default in _VM_DEFAULTS.items():
        data["vm"].setdefault(key, default)

    # Ensure provision.env exists
    data["provision"].setdefault("env", {})

    return data


def load_test_manifest(path: Path) -> dict:
    """Load a test manifest TOML file.

    Test manifests define install commands and test scenarios for a project.
    """
    with open(path, "rb") as f:
        data = tomllib.load(f)

    if "test" not in data:
        raise ValueError(f"Missing required section [test] in {path}")
    if "scenario" not in data:
        raise ValueError(f"No [[scenario]] entries in {path}")

    return data


def find_manifest(name: str, search_dirs: list[Path]) -> Path:
    """Find a VM manifest file by name across search directories.

    Looks for ``<name>.toml`` in each directory.
    Raises FileNotFoundError if not found.
    """
    for d in search_dirs:
        path = d / f"{name}.toml"
        if path.exists():
            return path
    raise FileNotFoundError(
        f"Manifest '{name}' not found in: {', '.join(str(d) for d in search_dirs)}"
    )
```

```python
"""Cloud-init provisioning for VMs."""
from __future__ import annotations

import subprocess
import textwrap
from pathlib import Path

import yaml


def generate_user_data(manifest: dict, ssh_pubkey: str) -> str:
    """Generate cloud-init user-data YAML from a VM manifest."""
    user = manifest["ssh"]["user"]
    prov = manifest["provision"]
    env = prov.get("env", {})

    # Build environment string for systemd unit
    env_lines = [f"Environment={k}={v}" for k, v in env.items()]
    env_block = "\n        ".join(env_lines) if env_lines else ""

    compositor_unit = textwrap.dedent(f"""\
        [Unit]
        Description=Test compositor
        [Service]
        Type=simple
        {env_block}
        ExecStart={prov['compositor_cmd']}
        [Install]
        WantedBy=default.target
    """).strip()

    data = {
        "users": [
            {
                "name": user,
                "ssh_authorized_keys": [ssh_pubkey],
                "sudo": "ALL=(ALL) NOPASSWD:ALL",
                "shell": "/bin/bash",
                "groups": ["video", "audio"],
            }
        ],
        "package_update": True,
        "packages": list(prov["packages"]),
        "write_files": [
            {
                "path": f"/home/{user}/.config/systemd/user/test-compositor.service",
                "owner": f"{user}:{user}",
                "content": compositor_unit,
            }
        ],
        "runcmd": [
            f"loginctl enable-linger {user}",
            f"systemctl --user -M {user}@ enable --now pipewire wireplumber || true",
            f"systemctl --user -M {user}@ enable --now test-compositor || true",
        ],
    }

    return "#cloud-config\n" + yaml.dump(data, default_flow_style=False)


def generate_meta_data(vm_name: str) -> str:
    """Generate cloud-init meta-data YAML."""
    data = {
        "instance-id": f"vmt-{vm_name}",
        "local-hostname": vm_name,
    }
    return yaml.dump(data, default_flow_style=False)


def create_cloud_init_iso(
    user_data: str, meta_data: str, output_path: Path
) -> None:
    """Write cloud-init user-data and meta-data, then generate an ISO.

    Uses ``cloud-localds`` (from cloud-image-utils) to create the ISO.
    """
    work_dir = output_path.parent
    ud_path = work_dir / "user-data"
    md_path = work_dir / "meta-data"

    ud_path.write_text(user_data)
    md_path.write_text(meta_data)

    subprocess.run(
        ["cloud-localds", str(output_path), str(ud_path), str(md_path)],
        check=True,
        capture_output=True,
    )

    # Clean up temp files
    ud_path.unlink(missing_ok=True)
    md_path.unlink(missing_ok=True)
```

```python
"""SSH client for VM communication."""
from __future__ import annotations

import logging
import time
from dataclasses import dataclass
from pathlib import Path

import paramiko

log = logging.getLogger(__name__)

_KEY_NAMES = ["id_ed25519", "id_rsa", "id_ecdsa"]


def get_ssh_key_path() -> Path:
    """Find the user's SSH private key.

    Checks for ed25519, RSA, and ECDSA keys in ~/.ssh/.
    Raises FileNotFoundError if no key is found.
    """
    ssh_dir = Path.home() / ".ssh"
    for name in _KEY_NAMES:
        path = ssh_dir / name
        if path.exists():
            return path
    raise FileNotFoundError(f"No SSH key found in {ssh_dir}")


def get_ssh_pubkey() -> str:
    """Read the user's SSH public key."""
    key_path = get_ssh_key_path()
    pub_path = key_path.with_suffix(".pub")
    if not pub_path.exists():
        raise FileNotFoundError(f"Public key not found: {pub_path}")
    return pub_path.read_text().strip()


@dataclass
class RunResult:
    """Result of a remote command execution."""
    stdout: str
    stderr: str
    returncode: int


class SSHClient:
    """SSH client for communicating with a VM."""

    def __init__(self, host: str, user: str, port: int = 22,
                 key_path: Path | None = None):
        self.host = host
        self.user = user
        self.port = port
        self.key_path = key_path or get_ssh_key_path()
        self._client: paramiko.SSHClient | None = None
        self._connected = False

    def connect(self) -> None:
        """Establish SSH connection to the VM."""
        self._client = paramiko.SSHClient()
        self._client.set_missing_host_key_policy(paramiko.AutoAddPolicy())
        self._client.connect(
            hostname=self.host,
            port=self.port,
            username=self.user,
            key_filename=str(self.key_path),
            timeout=10,
        )
        self._connected = True

    def close(self) -> None:
        """Close the SSH connection."""
        if self._client:
            self._client.close()
            self._connected = False

    def run(self, command: str) -> RunResult:
        """Execute a command on the remote VM and return the result."""
        if not self._connected or not self._client:
            self.connect()
        stdin, stdout, stderr = self._client.exec_command(command)
        exit_code = stdout.channel.recv_exit_status()
        return RunResult(
            stdout=stdout.read().decode(),
            stderr=stderr.read().decode(),
            returncode=exit_code,
        )

    def download(self, remote_path: str, local_path: Path) -> None:
        """Download a file from the VM via SFTP."""
        if not self._connected or not self._client:
            self.connect()
        sftp = self._client.open_sftp()
        try:
            local_path.parent.mkdir(parents=True, exist_ok=True)
            sftp.get(remote_path, str(local_path))
        finally:
            sftp.close()

    def upload(self, local_path: Path, remote_path: str) -> None:
        """Upload a file to the VM via SFTP."""
        if not self._connected or not self._client:
            self.connect()
        sftp = self._client.open_sftp()
        try:
            sftp.put(str(local_path), remote_path)
        finally:
            sftp.close()

    def wait_until_ready(self, timeout: float = 120, interval: float = 2) -> None:
        """Poll SSH until connection succeeds or timeout is reached."""
        deadline = time.time() + timeout
        last_err = None
        while time.time() < deadline:
            try:
                self.connect()
                log.info("SSH connection to %s ready", self.host)
                return
            except Exception as e:
                last_err = e
                log.debug("SSH not ready: %s, retrying in %ss", e, interval)
                time.sleep(interval)
        raise TimeoutError(
            f"SSH to {self.host}:{self.port} not ready after {timeout}s: {last_err}"
        )
```

```python
def _vm_dir(name: str) -> Path:

def generate_domain_xml(

class VMManager:
    """Manage VM lifecycle through libvirt."""

    def __init__(self, manifest_dirs: list[Path] | None = None):
        self._conn = libvirt.open("qemu:///system")
        if self._conn is None:
            raise RuntimeError("Failed to connect to libvirt")
        self._manifest_dirs = manifest_dirs or [
            Path(__file__).parent.parent / "manifests"
        ]

    def close(self) -> None:
        if self._conn:
            self._conn.close()

    def up(self, name: str) -> dict:
        """Boot a VM from a manifest. Returns info dict with IP, ports, etc."""
        manifest_path = find_manifest(name, self._manifest_dirs)
        manifest = load_vm_manifest(manifest_path)
        vm = manifest["vm"]
        workdir = _vm_dir(name)

        # 1. Download cloud image
        image_url = vm["image"]
        image_name = image_url.rsplit("/", 1)[-1]
        image_path = _IMAGES_DIR / image_name
        if not image_path.exists():
            log.info("Downloading %s ...", image_url)
            _IMAGES_DIR.mkdir(parents=True, exist_ok=True)
            urllib.request.urlretrieve(image_url, image_path)
            log.info("Downloaded to %s", image_path)

        # 2. Create overlay disk
        overlay_path = workdir / "disk.qcow2"
        self._create_overlay_disk(image_path, overlay_path)

        # 3. Generate cloud-init
        ssh_pubkey = get_ssh_pubkey()
        user_data = generate_user_data(manifest, ssh_pubkey)
        meta_data = generate_meta_data(name)
        ci_iso = workdir / "cloud-init.iso"
        create_cloud_init_iso(user_data, meta_data, ci_iso)

        # 4. Define and start domain
        xml = generate_domain_xml(
            name=name,
            memory_mb=vm["memory"],
            cpus=vm["cpus"],
            disk_path=str(overlay_path),
            cloud_init_iso=str(ci_iso),
        )
        domain_name = f"vmt-{name}"

        # Clean up existing domain if any
        try:
            old = self._conn.lookupByName(domain_name)
            if old.isActive():
                old.destroy()
            old.undefine()
        except libvirt.libvirtError:
            pass

        dom = self._conn.defineXML(xml)
        dom.create()
        log.info("Started VM %s", domain_name)

        # 5. Get IP and wait for SSH
        ip = self._wait_for_ip(dom, timeout=60)
        ssh = SSHClient(host=ip, user=manifest["ssh"]["user"],
                        port=manifest["ssh"]["port"])
        ssh.wait_until_ready(timeout=120)
        ssh.close()

        # 6. Get SPICE port
        spice_port = self._get_spice_port(dom)

        info = {
            "name": name,
            "domain": domain_name,
            "ip": ip,
            "ssh_user": manifest["ssh"]["user"],
            "ssh_port": manifest["ssh"]["port"],
            "spice_port": spice_port,
        }
        log.info("VM ready: %s (IP: %s, SPICE: %s)", name, ip, spice_port)
        return info

    def destroy(self, name: str) -> None:
        """Destroy a VM and clean up its files."""
        domain_name = f"vmt-{name}"
        try:
            dom = self._conn.lookupByName(domain_name)
            if dom.isActive():
                dom.destroy()
            dom.undefine()
        except libvirt.libvirtError as e:
            log.warning("Could not destroy %s: %s", domain_name, e)

        # Clean up working directory
        workdir = _vm_dir(name)
        for f in workdir.iterdir():
            f.unlink()
        workdir.rmdir()
        log.info("Destroyed VM %s", name)

    def get_info(self, name: str) -> dict | None:
        """Get info for a running VM, or None if not running."""
        domain_name = f"vmt-{name}"
        try:
            dom = self._conn.lookupByName(domain_name)
            if not dom.isActive():
                return None
            ip = self._get_ip(dom)
            spice_port = self._get_spice_port(dom)
            return {
                "name": name,
                "domain": domain_name,
                "ip": ip,
                "spice_port": spice_port,
            }
        except libvirt.libvirtError:
            return None

    def snapshot(self, name: str, snap_name: str) -> None:
        """Create a named snapshot of a VM."""
        domain_name = f"vmt-{name}"
        dom = self._conn.lookupByName(domain_name)
        xml = f"<domainsnapshot><name>{snap_name}</name></domainsnapshot>"
        dom.snapshotCreateXML(xml)
        log.info("Snapshot '%s' created for %s", snap_name, name)

    def restore(self, name: str, snap_name: str) -> None:
        """Revert a VM to a named snapshot."""
        domain_name = f"vmt-{name}"
        dom = self._conn.lookupByName(domain_name)
        snap = dom.snapshotLookupByName(snap_name)
        dom.revertToSnapshot(snap)
        log.info("Restored %s to snapshot '%s'", name, snap_name)

    def _create_overlay_disk(self, base: Path, overlay: Path) -> None:
        subprocess.run(
            ["qemu-img", "create", "-f", "qcow2",
             "-b", str(base), "-F", "qcow2", str(overlay)],
            check=True, capture_output=True,
        )

    def _wait_for_ip(self, dom, timeout: float = 60) -> str:
        """Wait for the VM to get an IP address via DHCP."""
        deadline = time.time() + timeout
        while time.time() < deadline:
            ip = self._get_ip(dom)
            if ip:
                return ip
            time.sleep(2)
        raise TimeoutError(f"VM {dom.name()} did not get an IP in {timeout}s")

    def _get_ip(self, dom) -> str | None:
        """Get VM IP from libvirt DHCP leases."""
        try:
            ifaces = dom.interfaceAddresses(
                libvirt.VIR_DOMAIN_INTERFACE_ADDRESSES_SRC_LEASE
            )
            for iface in ifaces.values():
                for addr in iface.get("addrs", []):
                    if addr["type"] == libvirt.VIR_IP_ADDR_TYPE_IPV4:
                        return addr["addr"]
        except libvirt.libvirtError:
            pass
        return None

    def _get_spice_port(self, dom) -> int | None:
        """Extract SPICE port from domain XML."""
        import xml.etree.ElementTree as ET
        xml = dom.XMLDesc()
        root = ET.fromstring(xml)
        graphics = root.find(".//graphics[@type='spice']")
        if graphics is not None:
            port = graphics.get("port")
            if port and port != "-1":
                return int(port)
        return None
```

```python
"""Screenshot capture, transfer, and comparison."""
from __future__ import annotations

import logging
from pathlib import Path

import numpy as np
from skimage.io import imread, imsave
from skimage.metrics import structural_similarity
from skimage.transform import resize

log = logging.getLogger(__name__)


def compare_screenshots(
    actual: Path,
    reference: Path,
    threshold: float = 0.95,
) -> tuple[bool, float]:
    """Compare two screenshots using structural similarity (SSIM).

    Returns (passed, score) where passed is True if score >= threshold.
    """
    img_a = imread(str(actual))
    img_b = imread(str(reference))

    # Handle different sizes (e.g. different display scales)
    if img_a.shape != img_b.shape:
        log.info(
            "Resizing reference from %s to %s", img_b.shape, img_a.shape
        )
        img_b = (resize(img_b, img_a.shape, anti_aliasing=True) * 255).astype(
            np.uint8
        )

    # Ensure both are RGB (drop alpha if present)
    if img_a.ndim == 3 and img_a.shape[2] == 4:
        img_a = img_a[:, :, :3]
    if img_b.ndim == 3 and img_b.shape[2] == 4:
        img_b = img_b[:, :, :3]

    score = structural_similarity(img_a, img_b, channel_axis=2)
    passed = score >= threshold
    log.info("SSIM: %.4f (threshold: %.2f) — %s", score, threshold,
             "PASS" if passed else "FAIL")
    return passed, float(score)


def generate_diff_image(
    actual: Path,
    reference: Path,
    output: Path,
) -> None:
    """Generate a visual diff image highlighting differences.

    Regions that differ are overlaid in red on the actual image.
    """
    img_a = imread(str(actual))
    img_b = imread(str(reference))

    if img_a.shape != img_b.shape:
        img_b = (resize(img_b, img_a.shape, anti_aliasing=True) * 255).astype(
            np.uint8
        )

    # Drop alpha
    if img_a.ndim == 3 and img_a.shape[2] == 4:
        img_a = img_a[:, :, :3]
    if img_b.ndim == 3 and img_b.shape[2] == 4:
        img_b = img_b[:, :, :3]

    # Compute per-pixel difference
    diff = np.abs(img_a.astype(int) - img_b.astype(int)).sum(axis=2)
    mask = diff > 30  # threshold for "different enough"

    # Overlay red on differing regions
    result = img_a.copy()
    result[mask] = [255, 0, 0]

    imsave(str(output), result)
    log.info("Diff image saved to %s (%d changed pixels)", output, mask.sum())
```

```python
from vmt.manifest import load_test_manifest, find_manifest, load_vm_manifest
from vmt.screenshot import compare_screenshots, generate_diff_image
from vmt.vm import VMManager


def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="vmt",
        description="Visual VM testing for Wayland applications",
    )
    parser.add_argument("-v", "--verbose", action="store_true")
    sub = parser.add_subparsers(dest="command")

    # vmt up <name>
    p_up = sub.add_parser("up", help="Boot a VM from a manifest")
    p_up.add_argument("name", help="VM manifest name (e.g. arch-sway)")

    # vmt destroy <name>
    p_destroy = sub.add_parser("destroy", help="Tear down a VM")
    p_destroy.add_argument("name")

    # vmt ssh <name> [-- command...]
    p_ssh = sub.add_parser("ssh", help="SSH into a VM or run a command")
    p_ssh.add_argument("name")
    p_ssh.add_argument("cmd", nargs="*", default=[], help="Command to run")

    # vmt view <name>
    p_view = sub.add_parser("view", help="Open SPICE viewer")
    p_view.add_argument("name")

    # vmt screenshot <name> <remote_path> <local_path>
    p_shot = sub.add_parser("screenshot", help="Download a screenshot from VM")
    p_shot.add_argument("name")
    p_shot.add_argument("remote_path")
    p_shot.add_argument("local_path")

    # vmt test <name> --manifest <path>
    p_test = sub.add_parser("test", help="Run test scenarios against a VM")
    p_test.add_argument("name")
    p_test.add_argument("--manifest", required=True, help="Path to test manifest")

    # vmt snapshot <name> <snap_name>
    p_snap = sub.add_parser("snapshot", help="Create a VM snapshot")
    p_snap.add_argument("name")
    p_snap.add_argument("snap_name")

    # vmt restore <name> <snap_name>
    p_restore = sub.add_parser("restore", help="Restore a VM snapshot")
    p_restore.add_argument("name")
    p_restore.add_argument("snap_name")

    # vmt update-references <name>
    p_ref = sub.add_parser("update-references", help="Copy latest screenshots as references")
    p_ref.add_argument("name")

    return parser


def _get_manager() -> VMManager:
    """Create a VMManager with the built-in manifest directory."""
    builtin = Path(__file__).parent.parent / "manifests"
    return VMManager(manifest_dirs=[builtin])


def cmd_up(args) -> int:
    mgr = _get_manager()
    try:
        info = mgr.up(args.name)
        print(f"VM {args.name} ready")
        print(f"  IP:    {info['ip']}")
        print(f"  SSH:   ssh {info['ssh_user']}@{info['ip']}")
        print(f"  SPICE: remote-viewer spice://127.0.0.1:{info['spice_port']}")
        return 0
    finally:
        mgr.close()


def cmd_destroy(args) -> int:
    mgr = _get_manager()
    try:
        mgr.destroy(args.name)
        print(f"VM {args.name} destroyed")
        return 0
    finally:
        mgr.close()


def cmd_ssh(args) -> int:
    mgr = _get_manager()
    try:
        info = mgr.get_info(args.name)
        if not info:
            print(f"VM {args.name} is not running", file=sys.stderr)
            return 1

        if args.cmd:
            # Non-interactive: run command and print output
            ssh = SSHClient(host=info["ip"], user=info.get("ssh_user", "root"),
                           port=info.get("ssh_port", 22))
            result = ssh.run(" ".join(args.cmd))
            if result.stdout:
                print(result.stdout, end="")
            if result.stderr:
                print(result.stderr, end="", file=sys.stderr)
            ssh.close()
            return result.returncode
        else:
            # Interactive: exec ssh
            import os
            os.execvp("ssh", [
                "ssh", "-o", "StrictHostKeyChecking=no",
                "-o", "UserKnownHostsFile=/dev/null",
                f"{info.get('ssh_user', 'root')}@{info['ip']}",
            ])
        return 0
    finally:
        mgr.close()


def cmd_view(args) -> int:
    mgr = _get_manager()
    try:
        info = mgr.get_info(args.name)
        if not info or not info.get("spice_port"):
            print(f"VM {args.name} is not running or has no SPICE", file=sys.stderr)
            return 1
        subprocess.Popen(
            ["remote-viewer", f"spice://127.0.0.1:{info['spice_port']}"],
            stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL,
        )
        print(f"Opened SPICE viewer for {args.name}")
        return 0
    finally:
        mgr.close()


def cmd_screenshot(args) -> int:
    mgr = _get_manager()
    try:
        info = mgr.get_info(args.name)
        if not info:
            print(f"VM {args.name} is not running", file=sys.stderr)
            return 1
        ssh = SSHClient(host=info["ip"], user=info.get("ssh_user", "root"),
                       port=info.get("ssh_port", 22))
        ssh.download(args.remote_path, Path(args.local_path))
        ssh.close()
        print(f"Downloaded {args.remote_path} -> {args.local_path}")
        return 0
    finally:
        mgr.close()


def cmd_test(args) -> int:
    mgr = _get_manager()
    try:
        info = mgr.get_info(args.name)
        if not info:
            print(f"VM {args.name} is not running", file=sys.stderr)
            return 1

        test_manifest = load_test_manifest(Path(args.manifest))
        ssh = SSHClient(host=info["ip"], user=info.get("ssh_user", "root"),
                       port=info.get("ssh_port", 22))

        # Load VM manifest to get screenshot_tool
        builtin = Path(__file__).parent.parent / "manifests"
        vm_manifest_path = find_manifest(args.name, [builtin])
        vm_manifest = load_vm_manifest(vm_manifest_path)
        screenshot_tool = vm_manifest["provision"].get("screenshot_tool", "grim")

        passed = 0
        failed = 0
        total = len(test_manifest["scenario"])

        for scenario in test_manifest["scenario"]:
            name = scenario["name"]

            # Run commands
            for cmd in scenario["commands"]:
                result = ssh.run(cmd)
                if result.returncode != 0:
                    log_line = result.stderr.strip() or f"exit code {result.returncode}"
                    print(f"  Warning: '{cmd}' failed: {log_line}")

            # Check expected output
            if "expect_output" in scenario:
                last_result = ssh.run(scenario["commands"][-1])
                if scenario["expect_output"] in last_result.stdout:
                    print(f"  ✓ {name}")
                    passed += 1
                else:
                    print(f"  ✗ {name} (expected '{scenario['expect_output']}', got '{last_result.stdout.strip()}')")
                    failed += 1
                continue

            # Screenshot comparison
            if "screenshot" in scenario:
                remote_path = f"/tmp/vmt-{scenario['screenshot']}"
                ssh.run(f"{screenshot_tool} {remote_path}")

                local_dir = Path(args.manifest).parent / "actual" / args.name
                local_dir.mkdir(parents=True, exist_ok=True)
                local_path = local_dir / scenario["screenshot"]
                ssh.download(remote_path, local_path)

                ref_path = Path(args.manifest).parent / scenario.get("reference", "")
                threshold = scenario.get("threshold", 0.95)

                if ref_path.exists():
                    ok, score = compare_screenshots(local_path, ref_path, threshold)
                    if ok:
                        print(f"  ✓ {name} (SSIM: {score:.2f})")
                        passed += 1
                    else:
                        diff_path = local_dir / f"{local_path.stem}-diff.png"
                        generate_diff_image(local_path, ref_path, diff_path)
                        print(f"  ✗ {name} (SSIM: {score:.2f}, diff: {diff_path})")
                        failed += 1
                else:
                    print(f"  ? {name} (no reference image at {ref_path})")
                    passed += 1  # No reference = informational
            else:
                print(f"  ✓ {name}")
                passed += 1

        ssh.close()
        print(f"\n{args.name}: {passed}/{total} passed" +
              (f", {failed} failed" if failed else ""))
        return 1 if failed else 0
    finally:
        mgr.close()


def cmd_snapshot(args) -> int:
    mgr = _get_manager()
    try:
        mgr.snapshot(args.name, args.snap_name)
        print(f"Snapshot '{args.snap_name}' created for {args.name}")
        return 0
    finally:
        mgr.close()


def cmd_restore(args) -> int:
    mgr = _get_manager()
    try:
        mgr.restore(args.name, args.snap_name)
        print(f"Restored {args.name} to '{args.snap_name}'")
        return 0
    finally:
        mgr.close()


def cmd_update_references(args) -> int:
    print(f"update-references for {args.name}: not yet implemented")
    return 0


_COMMANDS = {
    "up": cmd_up,
    "destroy": cmd_destroy,
    "ssh": cmd_ssh,
    "view": cmd_view,
    "screenshot": cmd_screenshot,
    "test": cmd_test,
    "snapshot": cmd_snapshot,
    "restore": cmd_restore,
    "update-references": cmd_update_references,
}


def main(argv: list[str] | None = None) -> int:
    parser = build_parser()
    args = parser.parse_args(argv)

    if args.verbose:
        logging.basicConfig(level=logging.DEBUG)
    else:
        logging.basicConfig(level=logging.INFO)

    if not args.command:
        parser.print_help()
        return 0

    handler = _COMMANDS.get(args.command)
    if handler:
        return handler(args)

    parser.print_help()
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
